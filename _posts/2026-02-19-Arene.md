from random import randint
class Arene:

    def __init__(self,longueur=9,largeur=8,fichier="arene.txt",):
        self.fichier=fichier
        try:
            with open(fichier, "r") as f:
                self.grille = self.db_tab()
        except FileNotFoundError:
            self.longueur=longueur
            self.largeur=largeur
            self.gen()
            self.grille=self.db_tab()

    


    def gen(self):
        interieur=self.largeur-2
        with open(self.fichier , "w") as f :
            f.write("1"*self.largeur)
            f.write("\n")
            f.write("1"*self.largeur)
            f.write("\n")  
            for loop in range(self.longueur):
                f.write("1")
                for loop in range(interieur):
                    f.write(" ")
                f.write("1")
                f.write("\n")
            f.write("1"*self.largeur)
            f.write("\n")
            f.write("1"*self.largeur)
            f.write("\n")


    
    def db_tab(self):
        grille=[]
        with open(self.fichier , "r") as f :
            liste=f.readlines()
            for ligne in liste:
                l=list(ligne)
                if "\n"in l:
                    l.remove("\n")
                grille.append(l)
        return grille
    def afficher(self):
        for ligne in self.grille :
            for element in ligne:
                print(element,end="")
            print()


    def verification(self):
        if len(self.grille) < 3 or len(self.grille[0]) < 3 :
            return False
        hauteur=len(self.grille)
        largeur=len(self.grille[0])
        for ligne in self.grille:
            for c in ligne:
                if c != " " and c != "1":
                    return False
        for x in range(largeur):
            if self.grille[0][x] != "1" or self.grille[hauteur - 1][x] != "1":
                return False
        for y in range(1, hauteur - 1):
            if self.grille[y][0] != "1" or self.grille[y][largeur - 1] != "1":
                return False

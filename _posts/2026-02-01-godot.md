extends Node2D

var cnt=1
var xp=0
var lvl=1
const pi =3.1415
@export var speed=10
@export_range(20.0, 50.0, 5.0)
var flot: float = 20.0
var inventaire = []
var list2=[2,3,7,5]

@export var col1: Color = Color.WHITE
@export var bullet: PackedScene
@export var texture1:Texture
enum choix{choix_1,choix_2,choix_3}
@export var mon_choix:choix



@onready var hero = $player
@onready var hand=$player/Sprite2D
func _ready():
	ennemy_kill(14)
	ennemy_kill(200,"dague")
	ennemy_kill(90,"500 piece")
	for x in 10:
		inventaire.append(null)
	inventaire[0]="armure"
	inventaire[1]="dague"
	inventaire[3]=list2
	inventaire.append("piece d'or")
	print(inventaire)
	print(inventaire.count("armure"))
	print(inventaire.find("armure"))
	
	
func inversechiffre(chiffre1,chiffre2):
	var temp=chiffre2
	chiffre2=chiffre1
	chiffre1=temp
	
func ennemy_kill(valeur_xp=1,drop=null):
	xp+=valeur_xp
	if xp>100:
		xp=0
		lvl+=1
	print("XP: "+str(xp)+" ,LVL: "+str(lvl)+" ,LOOT "+str(drop))

func _process(delta):
	cnt+=1
	hand.rotation_degrees=cnt
	$player/Label.text=str(hero.position)
	
	


func _on_button_pressed():
	hero.position.x= hero.position.x +speed


func _on_button_gauche_pressed():
	hero.position.x=hero.position.x-speed

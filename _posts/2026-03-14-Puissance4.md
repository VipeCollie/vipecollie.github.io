import pygame as py



class Plateau:
    def __init__(self):
        self.grille=self.cree_grille()

        self.lignes=6
        self.colonnes=7


        self.largeur= 750
        self.hauteur=300

        self.taille_case = (self.hauteur * 0.8) // self.lignes


        self.largeur_plateau = self.colonnes * self.taille_case
        self.hauteur_plateau = self.lignes * self.taille_case


        self.decalage_x = (self.largeur - self.largeur_plateau) // 2
        self.decalage_y = (self.hauteur - self.hauteur_plateau) // 2

        self.BLEU=(0,0,255)
        self.GRIS_FONCE=(100,100,100)
        self.GRIS_CLAIR=(200,200,200)

        self.cree_grille()

        py.init()
        self.fenetre = py.display.set_mode((self.largeur, self.hauteur))

    def cree_grille(self):
        self.grille=[]
        for ligne in range(6):
            nouvelle_ligne=[]
            for colonne in range(7):
                nouvelle_ligne.append(0)
            self.grille.append(ligne)


    def afficher(self):
        self.fenetre.fill(self.BLEU)
        py.draw.rect(
			self.fenetre,
			self.GRIS_FONCE,
			(self.decalage_x, self.decalage_y, self.largeur_plateau, self.hauteur_plateau)
		)
        for ligne in range(self.lignes):
                for colonne in range(self.colonnes):
                    x = self.decalage_x + colonne * self.taille_case + self.taille_case // 2
                    y = self.decalage_y + ligne * self.taille_case + self.taille_case // 2
                    rayon = self.taille_case // 2 - 5
                    py.draw.circle(self.fenetre, self.GRIS_CLAIR, (x, y), rayon)
        py.display.update()
    def souris(self):
        for event in py.event.get():
            if event.type == py.MOUSEBUTTONDOWN:
                x=event.pos[0]
                y=event.pos[1]
                print("Position du clic :",x,y)
                for ligne in range(self.lignes):
                    for colonne in range(self.colonnes):
                        if x < self.decalage_x or x > self.largeur-self.decalage_x:
                            print("hors du plateau")
                        if self.decalage_x + (colonne+1) * self.taille_case // 2 >x>self.decalage_x + colonne * self.taille_case // 2:
                            print("colonne 1")



    def lancer(self):
        en_cours = True
        while en_cours:
            for event in py.event.get():
                if event.type == py.QUIT:
                    en_cours = False

            self.afficher()
            self.souris()
        py.quit()
		    		    

jeu = Plateau()
jeu.lancer()
